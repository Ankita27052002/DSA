class Solution {
public:
    int singleNumber(vector<int>& nums) {
        // property 1: xor of two same element is always 0
        // property 2: xor of a number with 0 will always be that number - a ^ 0 = a

        int xorr = 0;
        for(int i=0;i<nums.size();i++){
            xorr = xorr ^ nums[i];
        }

        return xorr;
    }
};
